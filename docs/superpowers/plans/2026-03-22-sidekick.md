# Just Ship Sidekick Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build an embeddable AI chat widget (Sidekick) that lets project admins create, search, and manage tickets directly from any website via natural language.

**Architecture:** Hybrid Snippet (~3KB Vanilla JS) + iframe pointing to a standalone Next.js route in the Board app. The snippet handles activation, split-view layout, and context bridging via postMessage. The iframe houses the full chat UI with conversation history, auth, and AI-powered ticket operations. Backend API routes in the Board handle conversations, messages, and AI orchestration with Claude Sonnet.

**Tech Stack:** Next.js 16 (App Router), React 19, TypeScript, Supabase (PostgreSQL + Auth), Anthropic SDK (Claude Sonnet), shadcn/ui, TanStack Query v5, Zod 4, Vanilla JS (Snippet)

**Spec:** `docs/superpowers/specs/2026-03-22-sidekick-design.md`

```sql
-- 011_sidekick.sql
-- Add slug to projects + Sidekick tables

-- 1. Add slug column to projects
ALTER TABLE public.projects ADD COLUMN slug TEXT;

-- 2. Generate slugs from existing project names (kebab-case)
UPDATE public.projects
SET slug = LOWER(REGEXP_REPLACE(REGEXP_REPLACE(name, '[^a-zA-Z0-9\s-]', '', 'g'), '\s+', '-', 'g'))
WHERE slug IS NULL;

-- 3. Make slug NOT NULL + unique per workspace
ALTER TABLE public.projects ALTER COLUMN slug SET NOT NULL;
CREATE UNIQUE INDEX idx_projects_workspace_slug ON public.projects(workspace_id, slug);

-- 4. Sidekick conversations
CREATE TABLE public.sidekick_conversations (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  workspace_id    UUID NOT NULL REFERENCES public.workspaces(id) ON DELETE CASCADE,
  project_id      UUID NOT NULL REFERENCES public.projects(id) ON DELETE CASCADE,
  user_id         UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  title           TEXT,
  page_url        TEXT,
  page_title      TEXT,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- 5. Sidekick messages
CREATE TABLE public.sidekick_messages (
  id                UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  conversation_id   UUID NOT NULL REFERENCES public.sidekick_conversations(id) ON DELETE CASCADE,
  role              TEXT NOT NULL CHECK (role IN ('user', 'assistant')),
  content           TEXT NOT NULL,
  context           JSONB,
  ticket_id         UUID REFERENCES public.tickets(id),
  search_results    JSONB,
  created_at        TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- 6. Indexes
CREATE INDEX idx_sidekick_conversations_user_project
  ON public.sidekick_conversations(user_id, project_id);
CREATE INDEX idx_sidekick_conversations_updated
  ON public.sidekick_conversations(updated_at DESC);
CREATE INDEX idx_sidekick_messages_conversation
  ON public.sidekick_messages(conversation_id, created_at);

-- 7. RLS
ALTER TABLE public.sidekick_conversations ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users see own conversations"
  ON public.sidekick_conversations FOR ALL
  USING (auth.uid() = user_id);

ALTER TABLE public.sidekick_messages ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users see own messages"
  ON public.sidekick_messages FOR ALL
  USING (
    conversation_id IN (
      SELECT id FROM public.sidekick_conversations WHERE user_id = auth.uid()
    )
  );

-- 8. Trigger: update conversation timestamp on new message
CREATE OR REPLACE FUNCTION public.update_sidekick_conversation_timestamp()
RETURNS TRIGGER AS $$
BEGIN
  UPDATE public.sidekick_conversations
  SET updated_at = NOW()
  WHERE id = NEW.conversation_id;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_sidekick_conversation_updated
  AFTER INSERT ON public.sidekick_messages
  FOR EACH ROW
  EXECUTE FUNCTION public.update_sidekick_conversation_timestamp();
```

**Files:**
- Create: `just-ship-board/src/lib/sidekick/auth.ts`

- [ ] **Step 1: Create auth helper**

This helper validates that the current user session is a member of the workspace that owns the given project.

```typescript
import { createClient } from "@/lib/supabase/server";

interface SidekickAuthResult {
  user_id: string;
  workspace_id: string;
  project_id: string;
}

export async function requireSidekickAuth(projectId: string): Promise<SidekickAuthResult> {
  const supabase = await createClient();

  const { data: { user } } = await supabase.auth.getUser();
  if (!user) throw new SidekickAuthError("Not authenticated", 401);

  // Load project and verify it exists
  const { data: project } = await supabase
    .from("projects")
    .select("id, workspace_id")
    .eq("id", projectId)
    .single();

  if (!project) throw new SidekickAuthError("Project not found", 404);

  // Verify user is a workspace member (RLS on workspace_members enforces this)
  const { data: member } = await supabase
    .from("workspace_members")
    .select("id")
    .eq("workspace_id", project.workspace_id)
    .eq("user_id", user.id)
    .single();

  if (!member) throw new SidekickAuthError("Not a workspace member", 403);

  return {
    user_id: user.id,
    workspace_id: project.workspace_id,
    project_id: project.id,
  };
}

export class SidekickAuthError extends Error {
  constructor(message: string, public status: number) {
    super(message);
  }
}
```

```typescript
import { SupabaseClient } from "@supabase/supabase-js";

interface ToolContext {
  supabase: SupabaseClient;
  user_id: string;
  workspace_id: string;
  project_id: string;
}

export async function createTicket(
  ctx: ToolContext,
  args: { title: string; description: string; tags: string[]; priority: string }
) {
  // number is auto-generated by DB trigger trg_set_ticket_number — do NOT set manually
  const { data: ticket, error } = await ctx.supabase
    .from("tickets")
    .insert({
      workspace_id: ctx.workspace_id,
      project_id: ctx.project_id,
      title: args.title,
      body: args.description,
      tags: args.tags,
      priority: args.priority || "medium",
      status: "backlog",
      created_by: ctx.user_id,
    })
    .select("id, number, title, status, priority, tags, created_at")
    .single();

  if (error) throw new Error(`Failed to create ticket: ${error.message}`);
  return ticket;
}

export async function searchTickets(
  ctx: ToolContext,
  args: { query: string }
) {
  const pattern = `%${args.query}%`;

  const { data: tickets, error } = await ctx.supabase
    .from("tickets")
    .select("number, title, status, tags, created_at")
    .eq("workspace_id", ctx.workspace_id)
    .or(`title.ilike.${pattern},body.ilike.${pattern}`)
    .order("created_at", { ascending: false })
    .limit(10);

  if (error) throw new Error(`Failed to search tickets: ${error.message}`);
  return tickets || [];
}

export async function listMyTickets(
  ctx: ToolContext,
  args: { status?: string }
) {
  let query = ctx.supabase
    .from("tickets")
    .select("number, title, status, tags, priority, created_at")
    .eq("workspace_id", ctx.workspace_id)
    .eq("created_by", ctx.user_id);

  if (args.status) {
    query = query.eq("status", args.status);
  }

  const { data: tickets, error } = await query
    .order("created_at", { ascending: false })
    .limit(20);

  if (error) throw new Error(`Failed to list tickets: ${error.message}`);
  return tickets || [];
}
```

```typescript
import Anthropic from "@anthropic-ai/sdk";
import { createServiceClient } from "@/lib/supabase/service";
import { createTicket, searchTickets, listMyTickets } from "./tools";
import type { SidekickMessage } from "@/lib/types";

const anthropic = new Anthropic();

const SYSTEM_PROMPT = `Du bist der Just Ship Sidekick. Du hilfst Projekt-Admins dabei, Feedback und Änderungswünsche als strukturierte Tickets zu erfassen.

Du erhältst:
- Eine Nachricht des Nutzers in natürlicher Sprache
- Den Kontext der Seite, auf der sich der Nutzer befindet
- Den bisherigen Gesprächsverlauf

Deine Fähigkeiten:
1. Tickets erstellen — verstehe was der Nutzer möchte, frage nach wenn nötig, erstelle ein strukturiertes Ticket
2. Tickets suchen — durchsuche bestehende Tickets nach Stichworten
3. Eigene Tickets anzeigen — zeige dem Nutzer seine erstellten Tickets

Regeln:
- Bevor du ein Ticket erstellst, prüfe ob ein ähnliches existiert (search_tickets). Weise auf Duplikate hin.
- Wenn die Anfrage zu vage ist, frage nach Details.
- Setze passende Tags: "feature" für neue Funktionalität, "bug" für Fehler, "improvement" für Verbesserungen.
- Integriere den Seitenkontext in die Ticket-Beschreibung (z.B. "Gemeldet auf: /settings/profile").
- Antworte in der Sprache des Nutzers.
- Sei freundlich und prägnant.`;

const TOOLS: Anthropic.Tool[] = [
  {
    name: "create_ticket",
    description: "Erstellt ein neues Ticket im Board. Nutze dies nachdem du die Anfrage verstanden und nach Duplikaten gesucht hast.",
    input_schema: {
      type: "object" as const,
      properties: {
        title: { type: "string", description: "Prägnanter Titel, max 80 Zeichen" },
        description: { type: "string", description: "Detaillierte Beschreibung in Markdown, inkl. Seitenkontext" },
        tags: { type: "array", items: { type: "string" }, description: "Labels: feature, bug, improvement + weitere" },
        priority: { type: "string", enum: ["low", "medium", "high"], description: "Priorität" },
      },
      required: ["title", "description", "tags", "priority"],
    },
  },
  {
    name: "search_tickets",
    description: "Durchsucht bestehende Tickets nach Stichworten. Nutze dies vor dem Erstellen eines Tickets um Duplikate zu finden.",
    input_schema: {
      type: "object" as const,
      properties: {
        query: { type: "string", description: "Suchbegriff" },
      },
      required: ["query"],
    },
  },
  {
    name: "list_my_tickets",
    description: "Zeigt die eigenen Tickets des Nutzers an.",
    input_schema: {
      type: "object" as const,
      properties: {
        status: { type: "string", enum: ["backlog", "ready_to_develop", "in_progress", "in_review", "done", "cancelled"], description: "Optional: nach Status filtern" },
      },
    },
  },
];

interface StreamContext {
  user_id: string;
  workspace_id: string;
  project_id: string;
  project_name: string;
  project_description: string | null;
  messages: SidekickMessage[];
  page_context?: { url: string; path: string; title: string };
}

export async function streamSidekickResponse(
  ctx: StreamContext,
  onDelta: (text: string) => void,
  onToolCall: (tool: string, args: Record<string, unknown>) => void,
  onToolResult: (tool: string, result: unknown) => void,
): Promise<{ fullText: string; ticket?: Record<string, unknown>; searchResults?: Record<string, unknown>[] }> {
  const supabase = createServiceClient();

  const toolCtx = {
    supabase,
    user_id: ctx.user_id,
    workspace_id: ctx.workspace_id,
    project_id: ctx.project_id,
  };

  // Build message history for Claude
  const claudeMessages: Anthropic.MessageParam[] = ctx.messages.map((m) => ({
    role: m.role as "user" | "assistant",
    content: m.role === "user" && m.context
      ? `[Seitenkontext: ${m.context.title} — ${m.context.url}]\n\n${m.content}`
      : m.content,
  }));

  // Add page context to system prompt
  let systemPrompt = SYSTEM_PROMPT;
  systemPrompt += `\n\nProjekt: ${ctx.project_name}`;
  if (ctx.project_description) systemPrompt += ` — ${ctx.project_description}`;
  if (ctx.page_context) {
    systemPrompt += `\nAktuelle Seite: ${ctx.page_context.title} (${ctx.page_context.url})`;
  }

  let fullText = "";
  let ticket: Record<string, unknown> | undefined;
  let searchResults: Record<string, unknown>[] | undefined;

  // Use anthropic.messages.stream() — NOT .create({stream: true})
  // .stream() returns a MessageStream with .finalMessage() support
  const stream = anthropic.messages.stream({
    model: "claude-sonnet-4-20250514",
    max_tokens: 1024,
    system: systemPrompt,
    tools: TOOLS,
    messages: claudeMessages,
  });

  stream.on("text", (text) => {
    fullText += text;
    onDelta(text);
  });

  const finalMessage = await stream.finalMessage();

  // Handle tool use blocks
  const toolUseBlocks = finalMessage.content.filter(
    (b): b is Anthropic.ToolUseBlock => b.type === "tool_use"
  );

  if (toolUseBlocks.length > 0) {
    const toolResults: Anthropic.ToolResultBlockParam[] = [];

    for (const toolUse of toolUseBlocks) {
      onToolCall(toolUse.name, toolUse.input as Record<string, unknown>);

      let result: unknown;
      try {
        switch (toolUse.name) {
          case "create_ticket":
            result = await createTicket(toolCtx, toolUse.input as Parameters<typeof createTicket>[1]);
            ticket = result as Record<string, unknown>;
            break;
          case "search_tickets":
            result = await searchTickets(toolCtx, toolUse.input as Parameters<typeof searchTickets>[1]);
            searchResults = result as Record<string, unknown>[];
            break;
          case "list_my_tickets":
            result = await listMyTickets(toolCtx, toolUse.input as Parameters<typeof listMyTickets>[1]);
            searchResults = result as Record<string, unknown>[];
            break;
        }
      } catch (err) {
        result = { error: (err as Error).message };
      }

      onToolResult(toolUse.name, result);
      toolResults.push({
        type: "tool_result",
        tool_use_id: toolUse.id,
        content: JSON.stringify(result),
      });
    }

    // Continue conversation with tool results — stream the follow-up
    const followUpStream = anthropic.messages.stream({
      model: "claude-sonnet-4-20250514",
      max_tokens: 1024,
      system: systemPrompt,
      tools: TOOLS,
      messages: [
        ...claudeMessages,
        { role: "assistant", content: finalMessage.content },
        { role: "user", content: toolResults },
      ],
    });

    followUpStream.on("text", (text) => {
      fullText += text;
      onDelta(text);
    });

    await followUpStream.finalMessage();
  }

  return { fullText, ticket, searchResults };
}

export async function generateConversationTitle(firstMessage: string): Promise<string> {
  const response = await anthropic.messages.create({
    model: "claude-sonnet-4-20250514",
    max_tokens: 60,
    messages: [
      {
        role: "user",
        content: `Generiere einen kurzen Titel (max 50 Zeichen) für eine Conversation die mit dieser Nachricht beginnt. Nur den Titel, nichts anderes.\n\nNachricht: "${firstMessage}"`,
      },
    ],
  });

  const text = response.content[0];
  return text.type === "text" ? text.text.slice(0, 50) : "Neue Conversation";
}
```

```typescript
import { NextRequest } from "next/server";
import { createClient } from "@/lib/supabase/server";
import { success, error, unauthorized } from "@/lib/api/error-response";
import { requireSidekickAuth, SidekickAuthError } from "@/lib/sidekick/auth";
import { createConversationSchema, listConversationsSchema } from "@/lib/validations/sidekick";

export async function GET(request: NextRequest) {
  try {
    const params = Object.fromEntries(request.nextUrl.searchParams);
    const parsed = listConversationsSchema.safeParse(params);
    if (!parsed.success) return error("VALIDATION_ERROR", parsed.error.issues.map(i => i.message).join(", "), 400);

    const { project_id, limit, cursor } = parsed.data;
    const auth = await requireSidekickAuth(project_id);

    const supabase = await createClient();
    let query = supabase
      .from("sidekick_conversations")
      .select("id, title, updated_at, page_url")
      .eq("project_id", project_id)
      .eq("user_id", auth.user_id)
      .order("updated_at", { ascending: false })
      .limit(limit + 1); // fetch one extra for cursor

    if (cursor) {
      const { data: cursorConv } = await supabase
        .from("sidekick_conversations")
        .select("updated_at")
        .eq("id", cursor)
        .single();
      if (cursorConv) {
        query = query.lt("updated_at", cursorConv.updated_at);
      }
    }

    const { data: conversations, error: dbError } = await query;
    if (dbError) return error("DB_ERROR", dbError.message, 500);

    const items = conversations?.slice(0, limit) || [];
    const next_cursor = conversations && conversations.length > limit
      ? conversations[limit - 1].id
      : undefined;

    return success({ items, next_cursor });
  } catch (err) {
    if (err instanceof SidekickAuthError) return error("AUTH_ERROR", err.message, err.status);
    return error("INTERNAL_ERROR", "An unexpected error occurred", 500);
  }
}

export async function POST(request: NextRequest) {
  try {
    const body = await request.json();
```

```typescript
import { NextRequest } from "next/server";
import { createClient } from "@/lib/supabase/server";
import { createServiceClient } from "@/lib/supabase/service";
import { success, error } from "@/lib/api/error-response";
import { SidekickAuthError } from "@/lib/sidekick/auth";
import { sendMessageSchema, listMessagesSchema } from "@/lib/validations/sidekick";
import { streamSidekickResponse, generateConversationTitle } from "@/lib/sidekick/ai";

export async function GET(
  request: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  try {
    const { id: conversationId } = await params;
    const searchParams = Object.fromEntries(request.nextUrl.searchParams);
    const parsed = listMessagesSchema.safeParse(searchParams);
    if (!parsed.success) return error("VALIDATION_ERROR", parsed.error.issues.map(i => i.message).join(", "), 400);

    const supabase = await createClient();

    // Verify ownership via RLS (user can only see own conversations)
    const { data: conversation } = await supabase
      .from("sidekick_conversations")
      .select("id, project_id")
      .eq("id", conversationId)
      .single();

    if (!conversation) return error("NOT_FOUND", "Conversation not found", 404);

    const { limit, cursor } = parsed.data;
    let query = supabase
      .from("sidekick_messages")
      .select("id, role, content, context, ticket_id, search_results, created_at, ticket:tickets(id, number, title, status, priority, tags)")
      .eq("conversation_id", conversationId)
      .order("created_at", { ascending: true })
      .limit(limit + 1);

    if (cursor) {
      const { data: cursorMsg } = await supabase
        .from("sidekick_messages")
        .select("created_at")
        .eq("id", cursor)
        .single();
      if (cursorMsg) {
        query = query.gt("created_at", cursorMsg.created_at);
      }
    }

    const { data: messages, error: dbError } = await query;
    if (dbError) return error("DB_ERROR", dbError.message, 500);

    const items = messages?.slice(0, limit) || [];
    const next_cursor = messages && messages.length > limit
      ? messages[limit - 1].id
      : undefined;

    return success({ items, next_cursor });
  } catch (err) {
    if (err instanceof SidekickAuthError) return error("AUTH_ERROR", err.message, err.status);
    return error("INTERNAL_ERROR", "An unexpected error occurred", 500);
  }
}

export async function POST(
  request: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  try {
    const { id: conversationId } = await params;
    const body = await request.json();
    const parsed = sendMessageSchema.safeParse(body);
    if (!parsed.success) return error("VALIDATION_ERROR", parsed.error.issues.map(i => i.message).join(", "), 400);

    const supabase = await createClient();

    // Verify conversation ownership + get project info
    const { data: conversation } = await supabase
      .from("sidekick_conversations")
      .select("id, project_id, workspace_id, user_id, title, project:projects(id, name, description, workspace_id)")
      .eq("id", conversationId)
      .single();

    if (!conversation) return error("NOT_FOUND", "Conversation not found", 404);

    const project = conversation.project as { id: string; name: string; description: string | null; workspace_id: string };

    // Save user message
    const serviceClient = createServiceClient();
    await serviceClient
      .from("sidekick_messages")
      .insert({
        conversation_id: conversationId,
        role: "user",
        content: parsed.data.content,
        context: parsed.data.context || null,
      });

    // Load conversation history
    const { data: history } = await supabase
      .from("sidekick_messages")
      .select("id, role, content, context")
      .eq("conversation_id", conversationId)
      .order("created_at", { ascending: true })
      .limit(50);

    // Stream AI response via SSE
    const encoder = new TextEncoder();
    const stream = new ReadableStream({
      async start(controller) {
        const send = (event: string, data: unknown) => {
          controller.enqueue(encoder.encode(`event: ${event}\ndata: ${JSON.stringify(data)}\n\n`));
        };

        try {
          const result = await streamSidekickResponse(
            {
              user_id: conversation.user_id,
              workspace_id: conversation.workspace_id,
              project_id: project.id,
              project_name: project.name,
              project_description: project.description,
              messages: (history || []) as Parameters<typeof streamSidekickResponse>[0]["messages"],
              page_context: parsed.data.context,
            },
            (text) => send("delta", { text }),
            (tool, args) => send("tool_call", { tool, args }),
            (tool, result) => send("tool_result", { tool, result }),
          );

          // Save assistant message
          const { data: savedMessage } = await serviceClient
            .from("sidekick_messages")
            .insert({
              conversation_id: conversationId,
              role: "assistant",
              content: result.fullText,
              ticket_id: result.ticket ? (result.ticket as { id: string }).id : null,
              search_results: result.searchResults || null,
            })
            .select("id")
            .single();

          // Generate conversation title if first exchange
          if (!conversation.title && history && history.length <= 1) {
            const title = await generateConversationTitle(parsed.data.content);
            await serviceClient
              .from("sidekick_conversations")
              .update({ title })
              .eq("id", conversationId);
          }

          send("done", {
            message_id: savedMessage?.id,
            ticket: result.ticket,
            search_results: result.searchResults,
          });
        } catch (err) {
          send("error", { message: (err as Error).message });
        } finally {
          controller.close();
        }
      },
    });

    return new Response(stream, {
      headers: {
        "Content-Type": "text/event-stream",
        "Cache-Control": "no-cache",
        Connection: "keep-alive",
      },
    });
  } catch (err) {
    if (err instanceof SidekickAuthError) return error("AUTH_ERROR", err.message, err.status);
    return error("INTERNAL_ERROR", "An unexpected error occurred", 500);
  }
}
```

```tsx
"use client";

import { useState, useRef, useEffect } from "react";
import { useQuery } from "@tanstack/react-query";
import { Send, Loader2 } from "lucide-react";
import { Button } from "@/components/ui/button";
import { Textarea } from "@/components/ui/textarea";
import { MessageBubble } from "./message-bubble";
import type { SidekickMessage, SidekickContext } from "@/lib/types";

interface ChatViewProps {
  conversationId: string | null;
  projectId: string;
  workspaceSlug: string;
  pageContext: SidekickContext | null;
  onConversationCreated?: (id: string) => void;
}

export function ChatView({
  conversationId,
  projectId,
  workspaceSlug,
  pageContext,
  onConversationCreated,
}: ChatViewProps) {
  const [input, setInput] = useState("");
  const [sending, setSending] = useState(false);
  const [streamingText, setStreamingText] = useState("");
  const [localMessages, setLocalMessages] = useState<SidekickMessage[]>([]);
  const messagesEndRef = useRef<HTMLDivElement>(null);

  const { data: serverMessages } = useQuery({
    queryKey: ["sidekick-messages", conversationId],
    queryFn: async () => {
      if (!conversationId) return [];
      const res = await fetch(`/api/sidekick/conversations/${conversationId}/messages`);
      const json = await res.json();
      return (json.data?.items || []) as SidekickMessage[];
    },
    enabled: !!conversationId,
  });

  const allMessages = [...(serverMessages || []), ...localMessages];

  useEffect(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [allMessages.length, streamingText]);

  // Reset local messages when conversation changes
  useEffect(() => {
    setLocalMessages([]);
    setStreamingText("");
  }, [conversationId]);

  const handleSend = async () => {
    if (!input.trim() || sending) return;
    const content = input.trim();
    setInput("");
    setSending(true);
    setStreamingText("");

    // Add user message locally
    const userMsg: SidekickMessage = {
      id: crypto.randomUUID(),
      conversation_id: conversationId || "",
      role: "user",
      content,
      context: pageContext,
      ticket_id: null,
      search_results: null,
      created_at: new Date().toISOString(),
    };
    setLocalMessages((prev) => [...prev, userMsg]);

    try {
      // Create conversation if needed
      let convId = conversationId;
      if (!convId) {
        const res = await fetch("/api/sidekick/conversations", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({
            project_id: projectId,
            page_url: pageContext?.url,
            page_title: pageContext?.title,
          }),
        });
        const json = await res.json();
        convId = json.data.id;
        onConversationCreated?.(convId);
      }

      // Send message and stream response
      const res = await fetch(`/api/sidekick/conversations/${convId}/messages`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ content, context: pageContext }),
      });

      const reader = res.body?.getReader();
      const decoder = new TextDecoder();
      let fullText = "";
      let ticket: SidekickMessage["ticket"] = undefined;
      let searchResults: SidekickMessage["search_results"] = null;

      if (reader) {
        while (true) {
          const { done, value } = await reader.read();
          if (done) break;

          const chunk = decoder.decode(value);
          const lines = chunk.split("\n");
          let eventType = "";

          for (const line of lines) {
            if (line.startsWith("event: ")) eventType = line.slice(7);
            if (line.startsWith("data: ")) {
              const data = JSON.parse(line.slice(6));
              switch (eventType) {
                case "delta":
                  fullText += data.text;
                  setStreamingText(fullText);
                  break;
                case "tool_result":
                  if (data.tool === "create_ticket") ticket = data.result;
                  if (data.tool === "search_tickets" || data.tool === "list_my_tickets") searchResults = data.result;
                  break;
                case "done":
                  if (data.ticket) ticket = data.ticket;
                  if (data.search_results) searchResults = data.search_results;
                  break;
              }
            }
          }
        }
      }

      // Replace streaming text with final assistant message
      setStreamingText("");
      const assistantMsg: SidekickMessage = {
        id: crypto.randomUUID(),
        conversation_id: convId || "",
        role: "assistant",
        content: fullText,
        context: null,
        ticket_id: ticket ? (ticket as { id: string }).id : null,
        ticket: ticket as SidekickMessage["ticket"],
        search_results: searchResults,
        created_at: new Date().toISOString(),
      };
      setLocalMessages((prev) => [...prev, assistantMsg]);
    } catch (err) {
      setStreamingText("");
      const errorMsg: SidekickMessage = {
        id: crypto.randomUUID(),
        conversation_id: conversationId || "",
        role: "assistant",
        content: "Da ist etwas schiefgelaufen. Bitte versuch es nochmal.",
        context: null,
        ticket_id: null,
        search_results: null,
        created_at: new Date().toISOString(),
      };
      setLocalMessages((prev) => [...prev, errorMsg]);
    } finally {
      setSending(false);
    }
  };

  const handleKeyDown = (e: React.KeyboardEvent) => {
    if (e.key === "Enter" && !e.shiftKey) {
      e.preventDefault();
      handleSend();
    }
  };

  return (
    <div className="flex flex-col h-full">
      {/* Messages */}
      <div className="flex-1 overflow-y-auto p-4 space-y-3">
        {allMessages.length === 0 && !streamingText && (
          <div className="flex items-center justify-center h-full">
            <p className="text-muted-foreground text-sm">
              Hey! Was brauchst du auf dieser Seite?
            </p>
          </div>
        )}
        {allMessages.map((msg) => (
          <MessageBubble key={msg.id} message={msg} workspaceSlug={workspaceSlug} />
        ))}
        {streamingText && (
          <div className="flex justify-start">
            <div className="max-w-[85%] rounded-2xl px-4 py-2.5 bg-muted text-foreground">
              <div className="text-sm prose prose-sm max-w-none dark:prose-invert">
                <ReactMarkdown>{streamingText}</ReactMarkdown>
              </div>
            </div>
          </div>
        )}
        {sending && !streamingText && (
          <div className="flex justify-start">
            <div className="rounded-2xl px-4 py-2.5 bg-muted">
              <Loader2 className="h-4 w-4 animate-spin text-muted-foreground" />
            </div>
          </div>
        )}
        <div ref={messagesEndRef} />
      </div>

      {/* Input */}
      <div className="border-t p-3">
        <div className="flex gap-2">
          <Textarea
            value={input}
            onChange={(e) => setInput(e.target.value)}
            onKeyDown={handleKeyDown}
            placeholder="Beschreibe was du brauchst..."
            className="min-h-[40px] max-h-[120px] resize-none"
            rows={1}
          />
          <Button
            size="icon"
            onClick={handleSend}
            disabled={!input.trim() || sending}
          >
            <Send className="h-4 w-4" />
          </Button>
        </div>
      </div>
    </div>
  );
}
```

```tsx
import { useState, useEffect } from "react";
import { createClient } from "@/lib/supabase/client";
import { Button } from "@/components/ui/button";
import { LogIn } from "lucide-react";

interface SidekickAuthProps {
  children: React.ReactNode;
  projectWorkspaceId: string;
}

export function SidekickAuth({ children, projectWorkspaceId }: SidekickAuthProps) {
  const [authenticated, setAuthenticated] = useState<boolean | null>(null);
  const [isMember, setIsMember] = useState<boolean | null>(null);

  useEffect(() => {
    checkAuth();

    // Listen for auth-complete from popup
    const handler = (event: MessageEvent) => {
      if (event.data?.type === "sidekick:auth-complete") {
        checkAuth();
      }
    };
    window.addEventListener("message", handler);
    return () => window.removeEventListener("message", handler);
  }, []);

  const checkAuth = async () => {
    const supabase = createClient();
    const { data: { user } } = await supabase.auth.getUser();

    if (!user) {
      setAuthenticated(false);
      return;
    }

    setAuthenticated(true);

    // Check workspace membership
    const { data: member } = await supabase
      .from("workspace_members")
      .select("id")
      .eq("workspace_id", projectWorkspaceId)
      .eq("user_id", user.id)
      .single();

    setIsMember(!!member);
  };

  const openLogin = () => {
    window.open(
      "/auth/sidekick",
      "sidekick-auth",
      "width=400,height=500,left=200,top=200"
    );
  };

  if (authenticated === null) {
    return (
      <div className="flex items-center justify-center h-screen">
        <p className="text-muted-foreground text-sm">Laden...</p>
      </div>
    );
  }

  if (!authenticated) {
    return (
      <div className="flex flex-col items-center justify-center h-screen gap-4">
        <div className="text-center space-y-2">
          <h2 className="text-lg font-semibold">Just Ship Sidekick</h2>
          <p className="text-sm text-muted-foreground">
            Melde dich an, um Tickets zu erstellen.
          </p>
        </div>
        <Button onClick={openLogin} className="gap-2">
          <LogIn className="h-4 w-4" />
          Mit Just Ship einloggen
        </Button>
      </div>
    );
  }

  if (isMember === false) {
    return (
      <div className="flex items-center justify-center h-screen">
        <p className="text-sm text-muted-foreground">
          Du bist kein Mitglied dieses Workspaces.
        </p>
      </div>
    );
  }

  return <>{children}</>;
}
```

```tsx
"use client";

import { useState, useEffect, useCallback } from "react";
import { useQueryClient } from "@tanstack/react-query";
import { PanelLeftClose, PanelLeftOpen } from "lucide-react";
import { Button } from "@/components/ui/button";
import { SidekickAuth } from "./sidekick-auth";
import { ConversationSidebar } from "./conversation-sidebar";
import { ChatView } from "./chat-view";
import { ContextBadge } from "./context-badge";
import type { SidekickContext } from "@/lib/types";

interface SidekickClientProps {
  project: {
    id: string;
    name: string;
    description: string | null;
    slug: string;
    workspace_id: string;
  };
  workspaceName: string;
  workspaceSlug: string;
}

export function SidekickClient({ project, workspaceName, workspaceSlug }: SidekickClientProps) {
  const [conversationId, setConversationId] = useState<string | null>(null);
  const [sidebarOpen, setSidebarOpen] = useState(true);
  const [pageContext, setPageContext] = useState<SidekickContext | null>(null);
  const queryClient = useQueryClient();

  // Listen for context updates from snippet
  useEffect(() => {
    const handler = (event: MessageEvent) => {
      if (event.data?.type === "sidekick:context") {
        setPageContext({
          url: event.data.url,
          path: event.data.path,
          title: event.data.title,
          viewport: event.data.viewport,
        });
      }
    };
    window.addEventListener("message", handler);
    return () => window.removeEventListener("message", handler);
  }, []);

  // Restore active conversation from localStorage
  useEffect(() => {
    const saved = localStorage.getItem("sidekick:conversation");
    if (saved) setConversationId(saved);
  }, []);

  // Persist active conversation
  useEffect(() => {
    if (conversationId) {
      localStorage.setItem("sidekick:conversation", conversationId);
    }
  }, [conversationId]);

  const handleNewChat = useCallback(() => {
    setConversationId(null);
    localStorage.removeItem("sidekick:conversation");
  }, []);

  const handleConversationCreated = useCallback((id: string) => {
    setConversationId(id);
    queryClient.invalidateQueries({ queryKey: ["sidekick-conversations"] });
  }, [queryClient]);

  return (
    <SidekickAuth projectWorkspaceId={project.workspace_id}>
      <div className="flex flex-col h-screen">
        {/* Header */}
        <div className="flex items-center gap-2 px-3 py-2 border-b bg-background">
          <Button
            variant="ghost"
            size="icon"
            className="h-8 w-8"
            onClick={() => setSidebarOpen(!sidebarOpen)}
          >
            {sidebarOpen ? (
              <PanelLeftClose className="h-4 w-4" />
            ) : (
              <PanelLeftOpen className="h-4 w-4" />
            )}
          </Button>
          <span className="font-semibold text-sm truncate">{project.name}</span>
          <div className="ml-auto">
            <ContextBadge url={pageContext?.url || null} title={pageContext?.title || null} />
          </div>
        </div>

        {/* Body */}
        <div className="flex flex-1 overflow-hidden">
          {sidebarOpen && (
            <div className="w-60 shrink-0">
              <ConversationSidebar
                projectId={project.id}
                activeConversationId={conversationId}
                onSelect={setConversationId}
                onNewChat={handleNewChat}
              />
            </div>
          )}
          <div className="flex-1">
            <ChatView
              conversationId={conversationId}
              projectId={project.id}
              workspaceSlug={workspaceSlug}
              pageContext={pageContext}
              onConversationCreated={handleConversationCreated}
            />
          </div>
        </div>
      </div>
    </SidekickAuth>
  );
}
```

```javascript
(function() {
  'use strict';

  // Read config from script tag
  var script = document.currentScript;
  if (!script) return;
  var projectSlug = script.getAttribute('data-project');
  if (!projectSlug) return;

  var BOARD_URL = script.src.replace('/sidekick.js', '');
  var STORAGE_KEY = '_sk_state';
  var IFRAME_WIDTH = 400;
  var ACTIVATED = false;
  var container = null;
  var iframe = null;

  // Restore state
  var state = JSON.parse(localStorage.getItem(STORAGE_KEY) || '{}');

  function activate() {
    if (ACTIVATED) {
      deactivate();
      return;
    }
    ACTIVATED = true;
    state.open = true;
    localStorage.setItem(STORAGE_KEY, JSON.stringify(state));

    // Wrap body content
    var bodyChildren = Array.from(document.body.childNodes);
    var wrapper = document.createElement('div');
    wrapper.id = '_sk_wrapper';
    wrapper.style.cssText = 'display:flex;height:100vh;width:100%;overflow:hidden;';

    var content = document.createElement('div');
    content.id = '_sk_content';
    content.style.cssText = 'flex:1;overflow:auto;min-width:0;';

    bodyChildren.forEach(function(child) {
      if (child.id !== '_sk_wrapper') content.appendChild(child);
    });

    container = document.createElement('div');
    container.id = '_sk_panel';
    container.style.cssText = 'width:' + IFRAME_WIDTH + 'px;height:100vh;border-left:1px solid #e5e7eb;position:relative;flex-shrink:0;';

    // Close button
    var closeBtn = document.createElement('button');
    closeBtn.innerHTML = '&times;';
    closeBtn.style.cssText = 'position:absolute;top:8px;right:8px;z-index:10;background:none;border:none;font-size:20px;cursor:pointer;color:#6b7280;line-height:1;';
    closeBtn.onclick = deactivate;
    container.appendChild(closeBtn);

    // iframe
    iframe = document.createElement('iframe');
    iframe.src = BOARD_URL + '/sidekick/' + projectSlug;
    iframe.style.cssText = 'width:100%;height:100%;border:none;';
    iframe.allow = 'clipboard-write';
    container.appendChild(iframe);

    wrapper.appendChild(content);
    wrapper.appendChild(container);
    document.body.appendChild(wrapper);
    document.body.style.overflow = 'hidden';

    // Send initial context after iframe loads
    iframe.onload = function() {
      sendContext();
    };
  }

  function deactivate() {
    ACTIVATED = false;
    state.open = false;
    localStorage.setItem(STORAGE_KEY, JSON.stringify(state));

    var wrapper = document.getElementById('_sk_wrapper');
    var content = document.getElementById('_sk_content');
    if (wrapper && content) {
      var children = Array.from(content.childNodes);
      children.forEach(function(child) {
        document.body.appendChild(child);
      });
      wrapper.remove();
    }
    document.body.style.overflow = '';
    container = null;
    iframe = null;
  }

  function sendContext() {
    if (!iframe || !iframe.contentWindow) return;
    iframe.contentWindow.postMessage({
      type: 'sidekick:context',
      url: window.location.href,
      path: window.location.pathname,
      title: document.title,
      viewport: { width: window.innerWidth, height: window.innerHeight }
    }, BOARD_URL);
  }

  // Keyboard shortcut: Ctrl+Shift+S
  document.addEventListener('keydown', function(e) {
    if (e.ctrlKey && e.shiftKey && e.key === 'S') {
      e.preventDefault();
      activate();
    }
  });

  // URL parameter: ?sidekick
  if (window.location.search.indexOf('sidekick') !== -1) {
    // Wait for DOM
    if (document.readyState === 'loading') {
      document.addEventListener('DOMContentLoaded', activate);
    } else {
      activate();
    }
  }

  // Auto-restore if was open
  if (state.open) {
    if (document.readyState === 'loading') {
      document.addEventListener('DOMContentLoaded', activate);
    } else {
      activate();
    }
  }

  // SPA navigation detection
  var origPushState = history.pushState;
  var origReplaceState = history.replaceState;

  history.pushState = function() {
    origPushState.apply(this, arguments);
    sendContext();
  };

  history.replaceState = function() {
    origReplaceState.apply(this, arguments);
    sendContext();
  };

  window.addEventListener('popstate', sendContext);
})();
```
